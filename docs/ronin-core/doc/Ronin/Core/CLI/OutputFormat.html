def self.included(command)
  command.option :output_format, short: '-F',
                                 value: {
                                   type:    [:text, :columns, :csv, :json, :ndjson],
                                   default: :text
                                 },
                                 desc: 'Sets the output format'
end

def output_csv_schema(object)
  case object
  when Hash
    object.values
  else
    if object.respond_to?(:as_csv)
      object.as_csv
    else
      Array(object)
    end
  end
end

def output_json_schema(object)
  if object.respond_to?(:as_json)
    object.as_json
  elsif object.respond_to?(:to_h)
    object.to_h
  elsif object.respond_to?(:to_a)
    object.to_a
  else
    object
  end
end

def print_value(value)
  case options[:output_format]
  when :text
    print_text_value(value)
  when :columns
    puts output_columns_schema(value).join(' ')
  when :csv
    print_csv_value(value)
  when :json
    raise(NotImplementedError,"must call #print_values when output format is :json")
  when :ndjson
    print_ndjson_value(value)
  else
    raise(NotImplementedError,"output format not supported: #{options[:output_format].inspect}")
  end
end

def print_values(values)
  if options[:output_format] == :json
    values = case values
             when Hash
               Hash[values.map { |key,value|
                 [key, output_json_schema(value)]
               }]
             when Enumerable
               values.map(&method(:output_json_schema))
             else
               raise(ArgumentError,"values must be a Hash or an Enumerable object: #{values.inspect}")
             end

    if stdout.tty?
      jj values
    else
      puts JSON.generate(values)
    end
  else
    values.each do |value|
      print_value(value)
    end
  end
end
